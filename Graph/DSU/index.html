struct DSU {
  vector p, sz;
  DSU(int n): p(n), sz(n,1){ iota(p.begin(), p.end(), 0); }
  int find(int x){ return p[x]==x ? x : p[x]=find(p[x]); } // path compression
  bool unite(int a, int b){
    a = find(a); b = find(b);
    if(a==b) return false;               // already same group
    if(sz[a] < sz[b]) swap(a,b);         // union by size
    p[b] = a; sz[a] += sz[b];
    return true;
  }
};
